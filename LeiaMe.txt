O postgres tem que ter um usuário com nome e senha "admin".

O nome do banco de dados mysql é vaga_cursos.

Criando a tabela de usuários:
CREATE TABLE usuario (
	nome VARCHAR(255) NOT NULL, 
	sobrenome VARCHAR(255) NOT NULL, 
	email VARCHAR(255) NOT NULL , 
	senha VARCHAR(255) NOT NULL, 
	PRIMARY KEY (email)
);

Criando a tabela de categorias:
CREATE TABLE categorias (
	nome VARCHAR(255),
	PRIMARY KEY (nome)
);

Criando a tabela de cursos:
CREATE TABLE cursos (
	titulo VARCHAR(255),
	autor VARCHAR(255) NOT NULL,
	categoria VARCHAR(255) NOT NULL,
	PRIMARY KEY (titulo),
	FOREIGN KEY (autor) REFERENCES usuario(email),
	FOREIGN KEY (categoria) REFERENCES categorias(nome)
);

Crianda tabela de comentários:
CREATE TABLE comentarios (
	id INT AUTO_INCREMENT,
	curso VARCHAR(255),
	autor VARCHAR(255),
	conteudo VARCHAR(255) NOT NULL,
	PRIMARY KEY (id),
	FOREIGN KEY (curso) REFERENCES cursos(titulo),
	FOREIGN KEY (autor) REFERENCES usuario(email)
) AUTO_INCREMENT = 0;
